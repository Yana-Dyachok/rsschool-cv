                      function isValidIP(str) {
                        let reg = /[A-z]/;
                        let array = str.split(".").filter((element) => {
                          for (let index = 0; index = element.length+1; index++) {
                            if (element[0] === "0" && element.length >= 2) element = "\n";
                          }
                          return (
                            element >= 0 &&
                            element != 256 &&
                            !(element === "") &&
                            !element.includes("\n") &&
                            !element.includes(" ") &&
                            !element.match(reg)
                          );
                        });
                        return array.length !== 4 ? false : true;
                      }
